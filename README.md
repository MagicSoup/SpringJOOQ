Simple Rest project

Using :
- Spring 5
- Spring Boot 2.3
- Google AutoValue 1.7
- JOOQ 3.13
- Flyway 6.4
Endpoints :
  - Authors :
    - localhost:8080/v1/author
    - localhost:8080/v1/author/{authorUuid}
  - Books :   
    - localhost:8080/v1/book
    - localhost:8080/v1/book/{bookUuid}
    - localhost:8080/v1/book/author/{authorUuid}
